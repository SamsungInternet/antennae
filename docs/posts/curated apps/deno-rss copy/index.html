const fetch = require("node-fetch");

const html = String.raw;

class post { async data() { const data = await fetch( "https://deno-rss.glitch.me/manifest.json" ).then((r) => r.json());

	data.baseurl = "https://deno-rss.glitch.me/";
	data.tags = ['app'];
	data.template = 'default';

	return data;
}

render(manifest) {
	return html`
		<h1><a href="${new URL(manifest.starturl, manifest.baseurl).href}">${manifest.name}</a></h1>
		<p>This app is for reading RSS feeds.</p>
	`;
}

} module.exports = post;
